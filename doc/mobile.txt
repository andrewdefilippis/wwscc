
mobile/  - returns list of series 
	[
		{
			"name": string
			"parentseries": string
			"archived": boolean
			"locked": boolean
			"mtime": float
		}
	]


mobile/<seriesname>  - returns a list of events
	[
		{
			"id": int
			"name": string
			"location": string
			"date": "YYYY-MM-DD"
			"regopened": "YYYY-MM-DD HH:MM:SS"
			"regclosed": "YYYY-MM-DD HH:MM:SS"

			"sponsor": string
			"host": string
			"chair": string
			"designer": string
			"notes": string

			"ispro": boolean
			"practice": boolean
			"runs": int
			"countedruns": int
			"totlimit": int
			"perlimit": int
			"courses": int
			"conepen": float
			"gatepen": float

		}
	]


mobile/<seriesname>/classes - returns a list of the classes
	[
		{
			"code": string
			"descrip": string
			"champtrophy": boolean
			"eventtrophy": boolean
			"classmultiplier": float
			"carindexed": boolean
			"classindex": string
			"countedruns": boolean
			"usecarflag": boolean
			"numorder": int
		}
	]

mobile/<seriesname>/indexes - returns a list of the indexes
	[
		{
			"code": string
			"value": float
			"descrip": string
		}
	]

mobile/<seriesname>/<eventid>/last?classcodes=<classcode>[,<classcode>,...] - returns a the last finish time for the class codes, * means any class
	[
		{
			classcode: string
			updated: float
			carid: int
		}
	]


mobile/<seriesname>/<eventid>/classlist/<carid> - returns the class results, given extra info for specified car
	[
		{
			"carid": int
			"firstname": string
			"lastname": string
			"sum": float
			"courses": int
			"position": int
			"diff":  float
			"trophy": boolean
			"indexstr": string
			"pospoints": int
			"diffpoints": float
			"label": (if present) "current","raw","old
		}
	]


mobile/<seriesname>/<eventid>/champlist/<carid> - returns the champ results, given extra info for specified car/driver
	[
		{
			"driverid": int
			"carid": int
			"firstname": string
			"lastname": string
			"events": int
			"position": int
			"points": int or float depending on series config
			"label": (if present) "current","raw","old
		}
	]



mobile/<seriesname>/<eventid>/topraw/<carid> - returns the top raw times, given extra info for specified car/driver
	[
		{
			"carid": int
			"name": string
			"toptime": float
			"position": int
			"classcode": string
			"courses": int
			"label": (if present) "current","raw","old
		}
	]



mobile/<seriesname>/<eventid>/topnet/<carid> - returns the top net times, given extra info for specified car/driver
	[
		{
			"carid": int
			"name": string
			"position": int
			"toptime": float
			"classcode": string
			"indexstr": string
			"indexcode": string
			"indexvalue": float
			"courses": int
			"label": (if present) "current","raw","old
		}
	]
